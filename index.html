// FESTIlog — Google Apps Script (multi-feuilles)
// ═══════════════════════════════════════════════

function getOrCreate(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function styleHeader(sheet, nbCols) {
  sheet.getRange(1,1,1,nbCols).setFontWeight('bold').setBackground('#0F1923').setFontColor('#FF6B2B');
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // ── SUPPRESSION MOUVEMENT ────────────────────────────────
    if (data.action === 'deleteMouvement') {
      var sm = ss.getSheetByName('Mouvements');
      if (sm && sm.getLastRow() > 1) {
        var rows = sm.getRange(2, 1, sm.getLastRow()-1, sm.getLastColumn()).getValues();
        for (var i = rows.length-1; i >= 0; i--) {
          // Cherche par date+heure (colonnes 1 et 2)
          if (String(rows[i][0]) === String(data.date||'') && String(rows[i][1]) === String(data.heure||'')) {
            sm.deleteRow(i+2);
            break;
          }
        }
      }
      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    }

    // ── RESET COMPLET DU SHEET ───────────────────────────────
    if (data.action === 'resetSheet') {
      var feuilles = ['Mouvements','Stocks Centraux','Conso Réelle Globale','Config'];
      // Supprimer aussi toutes les feuilles CONSO - ...
      ss.getSheets().forEach(function(sh){
        var n = sh.getName();
        if(n.indexOf('CONSO - ') === 0) feuilles.push(n);
      });
      feuilles.forEach(function(nom){
        var sh = ss.getSheetByName(nom);
        if(sh) sh.clearContents();
      });
      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    }

    // ── AJOUT MOUVEMENT ──────────────────────────────────────
    if (data.action === 'addMouvement') {
      var sm = getOrCreate(ss, 'Mouvements');
      var header = ['Date','Heure','Type','Référence','Unité','Format','Source','Destination','Fournisseur','Centrale','Quantité','Note','Appareil'];
      // Toujours vérifier/mettre à jour l'en-tête (même si la feuille existe)
      if (sm.getLastRow() === 0) {
        sm.appendRow(header);
        styleHeader(sm, header.length);
        sm.getRange('A:B').setNumberFormat('@STRING@');
      } else {
        // Mettre à jour l'en-tête si le nombre de colonnes a changé
        var currentHeader = sm.getRange(1, 1, 1, sm.getLastColumn()).getValues()[0];
        if (currentHeader.length < header.length) {
          sm.getRange(1, 1, 1, header.length).setValues([header]).setNumberFormat('@STRING@');
          styleHeader(sm, header.length);
        }
      }
      var m = data.mouvement;
      var row = [m.date||'',m.heure||'',m.type,m.ref,m.unite||'',m.fmt||'',m.standSrc||'',m.standDst||'',m.fournisseur||'',m.centrale||'',m.qty,m.note||'',m.appareil||''];
      var lastRow = sm.getLastRow() + 1;
      sm.getRange(lastRow, 1, 1, row.length).setNumberFormat('@STRING@').setValues([row]);
      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    }

    // ── RÉCUPÉRER MOUVEMENTS ─────────────────────────────────
    if (data.action === 'getMouvements') {
      var sheet = ss.getSheetByName('Mouvements');
      if (!sheet || sheet.getLastRow() <= 1)
        return ContentService.createTextOutput(JSON.stringify({ok:true,rows:[]})).setMimeType(ContentService.MimeType.JSON);
      var rows = sheet.getRange(2,1,sheet.getLastRow()-1,11).getValues();
      return ContentService.createTextOutput(JSON.stringify({ok:true,rows:rows})).setMimeType(ContentService.MimeType.JSON);
    }

    // ── SNAPSHOTS (stocks + conso) ───────────────────────────
    if (data.action === 'updateSnapshots') {
      var snap = data.snapshot;

      // Stocks Centraux
      if (snap && snap.stocks) {
        var sc = getOrCreate(ss, 'Stocks Centraux');
        sc.clearContents();
        sc.appendRow(['Référence','Format','Unité','Stock DERVAL','Stock BASE LOG','Stock Total','Màj']);
        styleHeader(sc, 7);
        snap.stocks.forEach(function(r){ sc.appendRow(r); });
      }

      // Conso Réelle Globale
      if (snap && snap.consoGlobale) {
        var cg = getOrCreate(ss, 'Conso Réelle Globale');
        cg.clearContents();
        cg.appendRow(snap.consoGlobaleHeader);
        styleHeader(cg, snap.consoGlobaleHeader.length);
        snap.consoGlobale.forEach(function(r){ cg.appendRow(r); });
      }

      // Une feuille par stand
      if (snap && snap.stands) {
        snap.stands.forEach(function(standData) {
          var nomFeuille = 'CONSO - ' + standData.nom;
          if (nomFeuille.length > 100) nomFeuille = nomFeuille.slice(0, 100);
          var sh = getOrCreate(ss, nomFeuille);
          sh.clearContents();
          sh.appendRow(['Référence','Format','Unité','Reçu','Ret. Central','Trf. Sortant','Conso. Réelle','Màj']);
          styleHeader(sh, 8);
          standData.rows.forEach(function(r){ sh.appendRow(r); });
        });
      }

      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    }

    // ── CONFIG ───────────────────────────────────────────────
    if (data.action === 'syncConfig') {
      var sc2 = getOrCreate(ss, 'Config');
      sc2.clearContents();
      sc2.appendRow(['produits_json', JSON.stringify(data.produits)]);
      sc2.appendRow(['stands_json', JSON.stringify(data.stands)]);
      sc2.appendRow(['formats_json', JSON.stringify(data.formats)]);
      return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
    }

    if (data.action === 'getConfig') {
      var sc3 = ss.getSheetByName('Config');
      if (!sc3 || sc3.getLastRow() === 0)
        return ContentService.createTextOutput(JSON.stringify({ok:true,config:null})).setMimeType(ContentService.MimeType.JSON);
      var cfgRows = sc3.getDataRange().getValues();
      var cfg = {};
      cfgRows.forEach(function(r){ try{ cfg[r[0]]=JSON.parse(r[1]); }catch(ex){} });
      return ContentService.createTextOutput(JSON.stringify({ok:true,config:cfg})).setMimeType(ContentService.MimeType.JSON);
    }

  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e){ return doPost(e); }
